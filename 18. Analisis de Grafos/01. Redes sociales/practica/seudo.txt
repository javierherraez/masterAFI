estado nods = S susceptible (map para nodos set_vertex_attr("state", value = 'S'))

seleccionar N nodos aleatorio: state = I infectado

n_contagiados = N
n_recuperados = 0
dias_sin_contagios = 0

While dias_sin_contagios < 3:

	n_contagiados_iniciales = n_contagiados
	n_contagiados_aux = n_contagiados

	for nodo in grafo:

		if nodo == infectado:
			-> intento de recuperacion
			if nodo == recuperado: n_recuperados ++; n_contagiados --; 

		if nodo == susceptible:
			for vecino in vecinos(nodo):
				 if vecino == infectado:
				 	-> intento de contagio
				 	if nodo == contagiado: n_contagiados ++; n_susceptibles--; n_contagiados_aux ++; break

		if nodo == recuperado: next

	if n_contagiados_aux == n_contagiados_iniciales
		-> dias_sin_contagios ++
