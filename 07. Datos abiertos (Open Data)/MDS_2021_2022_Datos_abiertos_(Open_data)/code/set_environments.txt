# Beautifulsup Eurostoxx
conda remove env --name beautifulsoup_eurostoxx --all
conda create -n beautifulsoup_eurostoxx python=3.8.11
conda activate beautifulsoup_eurostoxx
conda install numpy beautifulsoup4 pandas requests lxml

# Scrapy 20minutos
conda remove env --name scrapy_20minutos --all
conda create -n scrapy_20minutos python=3.8.11 
conda activate scrapy_20minutos
conda install scrapy protego

scrapy startproject scraper_20minutos
scrapy genspider -t crawl spider 20minutos.es
scrapy crawl 20minutos -O news.json
