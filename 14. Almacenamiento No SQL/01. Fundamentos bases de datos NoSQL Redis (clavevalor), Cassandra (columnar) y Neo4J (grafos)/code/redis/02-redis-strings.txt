# Establecimiento el valor de una clave de tipo String
SET mensaje hola
SET mensaje 'hola mundo'
SET prueba 1
SET usuario:1 mcorella
SET 'usuario 2' mcorella
MSET usuario:3 jsmith usuario:4 jdoe
SETNX mensaje 'mundo'


# Recueración de valor de clave
GET usuario:1
MGET mensaje usuario:2 usuario:3


# Recuperación de la longitud de una clave
STRLEN mensaje


# Recuperación parcial del contenido de una clave
GETRANGE usuario:1 1 1
GETRANGE usuario:1 1 -1


# Concatenación de contenido a una clave
APPEND mensaje ' mundo'


# Incremento de valor de una clave "numérica"
INCR prueba
INCRBY prueba 10



# Decremento de valor de una clave "numérica"
DECR prueba
DECRBY prueba 10
