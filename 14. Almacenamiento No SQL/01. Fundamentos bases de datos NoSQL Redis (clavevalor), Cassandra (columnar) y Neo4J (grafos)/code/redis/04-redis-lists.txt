# Inserción de elementos en una lista por los lados
LPUSH lista 0
RPUSH lista 10
RPUSH lista 11


# Recuperación de elementos en una lista
LRANGE lista 0 0
LRANGE lista 0 -1


# Inserción de elementos en base a pivote
LINSERT lista BEFORE 6 5
LINSERT lista AFTER 10 10.5


# Recuperación (y eliminación) de elementos por los lados
LPOP lista
RPOP lista


# Eliminación de elementos de una lista por valor
LREM lista 1 10.5


# Modificaciónd de valor por posición
LSET lista 0 25


# Longitud de una lista
LLEN lista


# Recuperación de un elemento por posición
LPUSH lista 15
LINDEX lista 1
