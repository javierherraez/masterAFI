# Claves de prueba
SET prueba 1
SET mensaje hola
SET usuario:1 mcorella

# Eliminación de claves
DEL prueba
DEL mensaje

# Chequeo sobre la existencia de una clave
EXISTS usuario:1


# Establecimiento de un tiempo de expiración a una clave
SET contador 0
EXPIRE contador 10
PEXPIRE contador 3000


# Recuperación del tiempo restante de vida de una clave
SET contador 0
EXPIRE contador 20
TTL contador
PTTL contador


# Recuperación de todas las claves disponibles (en un patrón)
KEYS *
KEYS usuario:*


# Modificación del nombre de una clave
SET contador 0
RENAME contador num.usuarios
RENAMENX num.usuarios usuario:1



# Recuperación del tipo de una clave
TYPE usuario:1


# Eliminación de todas las claves de la base de datos
FLUSHDB
