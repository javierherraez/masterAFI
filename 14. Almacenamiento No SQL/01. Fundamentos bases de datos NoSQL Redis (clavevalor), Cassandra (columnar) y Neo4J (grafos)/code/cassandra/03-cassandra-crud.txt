# Inserción de contenido en una tabla

INSERT INTO tabla_prueba (id, nombre, activo) VALUES (0, 'mcorella', false);
INSERT INTO tabla_prueba (id, nombre, activo) VALUES (0, 'jsmith', true);
INSERT INTO tabla_prueba (id, nombre, activo) VALUES (1, 'mcorella', false);


# Recuperación de contenido de una tabla
SELECT * FROM tabla_prueba;
SELECT id, nombre FROM tabla_prueba;
SELECT id, nombre FROM tabla_prueba WHERE id = 0;
SELECT id, nombre FROM tabla_prueba WHERE nombre = 'mcorella';


# Creación de índices sobre columnas de una tabla
CREATE INDEX tabla_prueba_nombre ON tabla_prueba (nombre);
CREATE INDEX tabla_prueba_activo ON tabla_prueba (activo);


# Recuperación de contenido de una tabla filtrando por campos no primarios
SELECT id, nombre FROM tabla_prueba WHERE activo = true;
SELECT id, nombre FROM tabla_prueba WHERE nombre = 'mcorella';
SELECT id, nombre FROM tabla_prueba WHERE activo = true AND nombre = 'mcorella';


# Elminación de índices
DROP INDEX tabla_prueba_nombre;
DROP INDEX tabla_prueba_activo;


# Actualización de datos
UPDATE tabla_prueba SET nombre = 'mcorella1' WHERE id = 0;


# Eliminación de datost()
DELETE tabla_prueba WHERE id = 0;
