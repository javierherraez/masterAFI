// Creamos algunos nodos y relaciones sobre los que trabajar
CREATE 
	p = 
	(:Profesor { name: 'Miguel Ángel Corella'})
	-[:IMPARTE]->
	(:Materia { name: 'BBDD NoSQL'})
	<-[:ASISTE]-
	(:Promocion { year: '2018-2019'})
RETURN p;

// Añadimos propiedades a nodos
MATCH (miguel:Profesor { name: 'Miguel Ángel Corella'})
SET miguel.age = 36;

MATCH (nosql:Materia { name: 'BBDD NoSQL'})
SET nosql.quarter = 'Q2', nosql.homework = false;

// Añadimos propiedades a las relaciones
MATCH ()-[rel:IMPARTE]-()
SET rel.schedule = '17:00 - 21:00';

MATCH ()-[rel:ASISTE]-()
SET rel.attendants = 35;

// Añadimos etiquetas a nodos
MATCH (miguel:Profesor { name: 'Miguel Ángel Corella'})
SET miguel:Persona;


// ELIMINACION DE PROPIEDADES
//--------------------------------

// Eliminamos propiedades de un nodo
MATCH (miguel:Profesor { name: 'Miguel Ángel Corella'})
REMOVE miguel.age;

MATCH (nosql:Materia { name: 'BBDD NoSQL'})
REMOVE nosql.quarter, nosql.homework;

// Eliminamos propiedades a las relaciones
