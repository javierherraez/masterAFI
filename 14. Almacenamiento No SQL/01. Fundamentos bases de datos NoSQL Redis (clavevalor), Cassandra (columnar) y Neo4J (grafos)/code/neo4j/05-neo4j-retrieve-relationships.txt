// RELACIONES
//----------------------

// Recuperación de todas las relaciones del grafo
MATCH ()-[rel]-() RETURN rel;

// Recuperación de todas las relaciones del grafo limitando los resultados
MATCH ()-[rel]-() RETURN rel LIMIT 10;

// Recuperación de todas las relaciones del grafo con una determinada dirección
MATCH ()-[rel]->() RETURN rel;

// Recuperación de todas las relaciones del grafo con un determinado tipo
MATCH ()-[rel:ACTED_IN] RETURN rel;

// Recuperación de todas las relaciones del grafo con un determinado valor de propiedad
MATCH ()-[rel {rating: 65}] RETURN rel;

// Recuperación de tipos de las relaciones
MATCH ()-[rel]-() RETURN type(n);

// Recuperación de propiedades de nodos
MATCH ()-[rel:REVIEWED]-() RETURN rel.rating;

// Recuperación y alias de propiedades de nodos
MATCH ()-[rel:REVIEWED]-() RETURN rel.rating as Puntuación;
