// NODOS
//----------------------

// Creación de un nodo sencillo
CREATE ();

// Comprobamos que el nodo se ha creado con éxito
MATCH (n) RETURN n;

// Creación de múltiples nodos
CREATE (), ();

// Comprobamos que los nodos se han creado con éxito
MATCH (n) RETURN n;

// Creamos un nodo con una etiqueta identificadora
CREATE (:Profesor);

// Comprobamos que el nodo se ha creado con éxito
MATCH (n) RETURN n;

// Creamos un nodo con múltiples etiquetas
CREATE (:Profesor:Persona);

// Comprobamos que el nodo se ha creado con éxito
MATCH (n) RETURN n;

// Creamos un nodo con propiedades
CREATE (:Profesor:Persona { name: 'Miguel Ángel', age: 36});

// Comprobamos que el nodo se ha creado con éxito
MATCH (n) RETURN n;

// Creamos un nodo e incluimos el nodo creado en el retorno
CREATE (miguel:Profesor:Persona { name: 'Miguel Ángel', age: 36}) RETURN miguel;
