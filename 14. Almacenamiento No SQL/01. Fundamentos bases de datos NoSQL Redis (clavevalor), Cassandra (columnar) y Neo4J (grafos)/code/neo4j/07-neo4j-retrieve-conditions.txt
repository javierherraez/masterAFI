// CONDICIONES
//----------------------

// Filtrado de nodos por valor de propiedad
MATCH (node:Movie) 
WHERE node.title = 'The Matrix'
RETURN node;

// Filtrado de relaciones por valor de propiedad
MATCH ()-[rel:REVIEWED]- () 
WHERE rel.rating = 65
RETURN rel;

// Filtrado múltiple 
MATCH (movie:Movie)<--(person:Person)
WHERE movie.title = 'The Matrix' AND person.name = 'Keanu Reeves'
RETURN movie.title, person.name;

MATCH (movie:Movie)<--(person:Person)
WHERE movie.title = 'The Matrix' OR person.name = 'Keanu Reeves'
RETURN movie.title, person.name;

MATCH (movie:Movie)<--(person:Person)
WHERE movie.title = 'The Matrix' XOR person.name = 'Keanu Reeves'
RETURN movie.title, person.name;

MATCH (movie:Movie)<--(person:Person)
WHERE movie.title = 'The Matrix' AND NOT person.name = 'Keanu Reeves'
RETURN movie.title, person.name;

// Filtrado por existencia de campo
MATCH (node)
WHERE exists(node.name)
RETURN node;

// Filtrado por múltiples valores
MATCH (node:Movie)-[:ACTED_IN]-(person:Person)
WHERE node.title IN ['The Matrix', 'That Thing You Do']
RETURN person.name;

// Filtrado por texto
MATCH (node:Movie)
WHERE node.title STARTS WITH 'That'
RETURN node;

MATCH (node:Movie)
WHERE node.title ENDS WITH 'Revolution'
RETURN node;

MATCH (node:Movie)
WHERE node.title CONTAINS 'Matrix'
RETURN node;
